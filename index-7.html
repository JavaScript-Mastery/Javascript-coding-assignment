Que.2 Delete a Parameter

  function Check(obj1, rollno) {
    // Check if the object has the `rollno` property
    if (obj1.hasOwnProperty("rollno") && obj1.rollno === rollno) {
      // Delete the `rollno` property
      delete obj1.rollno;
      return true;
    } else {
      return false;
    }   
  }
