Que.1 Calculator

            
            const Calculator = (A, B, C) => 
            {
              switch (A) {
                case "+":
                  return B + C;
                case "-":
                  return B - C;
                case "*":
                  return B * C;
                case "/":
                  return B / C;
                default:
                  return "Invalid operator";
              }
                
            };
